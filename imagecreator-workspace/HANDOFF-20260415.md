
### P1 · 换 HTTP 客户端（与 crawler 同步做）

**最好和 crawler 的 urllib→httpx 改造在同一次会话里做，或者紧接着做。** 两个项目的 tg() 函数结构类似：

```python
# imagecreator/bin/tg_bot.py (lines 233-260) 改写目标
import httpx

_client = httpx.Client(
    timeout=httpx.Timeout(connect=10, read=70, write=10, pool=10),
    http2=False,
    follow_redirects=False,
)

def _tg_get(method, params=None, timeout=15):
    url = TG_API + "/" + method
    try:
        r = _client.get(url, params=params, timeout=timeout)
        return r.json()
    except httpx.HTTPError as e:
        print(f"[tg_get] {method} error: {e}", flush=True)
        return {}

def _tg_post_json(method, data, timeout=15):
    url = TG_API + "/" + method
    try:
        r = _client.post(url, json=data, timeout=timeout)
        return r.json()
    except httpx.HTTPError as e:
        # preserve error body for Telegram description
        if hasattr(e, 'response') and e.response is not None:
            try: return e.response.json()
            except: pass
        return {"ok": False, "error_code": None, "description": str(e)}
    except Exception as e:
        return {"ok": False, "error": str(e)}
```
